CREATE TABLE Raktár
(
  Raktár_id INT NOT NULL,
  Ra_Darabszám INT NOT NULL,
  Raktár_Ár INT NOT NULL,
  Csomag_tipus Varchar(30) NOT NULL,
  PRIMARY KEY (Raktár_id)
);

CREATE TABLE Vevő
(
  Irányítószám INT NOT NULL,
  Település Varchar(30) NOT NULL,
  Házszám INT NOT NULL,
  Utca Varchar(30) NOT NULL,
  email Varchar(30) NOT NULL,
  Név Varchar(30) NOT NULL,
  PRIMARY KEY (Név)
);

CREATE TABLE Alkalmazott
(
  Alkalmazott_név Varchar(30) NOT NULL,
  Alkalmazott_Fizetés INT NOT NULL,
  Beosztás Varchar(30) NOT NULL,
  Csatlakozás Date NOT NULL,
  PRIMARY KEY (Csatlakozás)
);
CREATE TABLE Raktár_Alkalmazott
(
  Elérhető Varchar(30) NOT NULL,
  Raktár_id INT NOT NULL,
  Csatlakozás Date NOT NULL,
  PRIMARY KEY (Raktár_id, Csatlakozás),
  FOREIGN KEY (Raktár_id) REFERENCES Raktár(Raktár_id),
  FOREIGN KEY (Csatlakozás) REFERENCES Alkalmazott(Csatlakozás)
);

CREATE TABLE Vevő_Telefonszám
(
  Telefonszám INT NOT NULL,
  Név Varchar(30) NOT NULL,
  PRIMARY KEY (Telefonszám, Név),
  FOREIGN KEY (Név) REFERENCES Vevő(Név)
);

CREATE TABLE Rendelés
(
  Rendelés_id INT NOT NULL,
  Rendelés_ár INT NOT NULL,
  Dátum Date NOT NULL,
  Rendelés_Fizetés INT NOT NULL,
  Név Varchar(30) NOT NULL,
  PRIMARY KEY (Rendelés_id),
  FOREIGN KEY (Név) REFERENCES Vevő(Név)
);

CREATE TABLE Termék
(
  Cipő Varchar(30) NOT NULL,
  Ruha Varchar(30) NOT NULL,
  Kiegészítő Varchar(30) NOT NULL,
  Termék_id INT NOT NULL,
  Rendelés_id INT NOT NULL,
  PRIMARY KEY (Termék_id),
  FOREIGN KEY (Rendelés_id) REFERENCES Rendelés(Rendelés_id)
);

CREATE TABLE Raktár_Rendelés
(
  Rendelés_id INT NOT NULL,
  Raktár_id INT NOT NULL,
  PRIMARY KEY (Rendelés_id, Raktár_id),
  FOREIGN KEY (Rendelés_id) REFERENCES Rendelés(Rendelés_id),
  FOREIGN KEY (Raktár_id) REFERENCES Raktár(Raktár_id)
);
